CREATE TABLE Produto (
    CodigoProduto int IDENTITY not null,
    Nome varchar(20),
    NomeDescritivo varchar(20),
	DataDeCriacao date,
	LocalNoEstoque varchar(40),
	Quantidade int,
	Preco decimal (18,2),
	
	constraint PK_CodigoProduto Primary Key (CodigoProduto)
)
